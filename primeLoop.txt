public class primeLoop {
    public static void findPrimeNumber(Integer lowerLimit, Integer upperLimit){
        for(Integer i = lowerLimit; i<= upperLimit; i++){
            Boolean isPrime = true;
            for(Integer j =2; j <= Math.sqrt(i); j++){
                if(Math.mod(i, j) == 0){
                    isPrime = false;
                    break;
                }
            }
            if(isPrime && i>1){
                System.debug(i);
            }
        }
    }
    
}
